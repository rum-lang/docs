    printLn result

4.3. Fizz-Buzz (v1)¶

2

fizzBuzz n = n
    | n % 15 is 0 => "FizzBuzz"
    | n % 3 is 0  => "Fizz"
    | n % 5 is 0  => "Buzz"
    | else        => n

main =
    printLn fizzBuzz [0..100]
